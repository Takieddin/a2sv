# The isBadVersion API is already defined for you.
# def isBadVersion(version: int) -> bool:

class Solution:
    def firstBadVersion(self, n: int) -> int:
        b, e = 1, n
        while b <= e:
            m = (b + e) // 2
            if  isBadVersion(m):
                if m == 1:
                    return 1
                if not isBadVersion(m - 1):
                    return m
                e = m - 1
            else :
                b = m + 1
        return -1
