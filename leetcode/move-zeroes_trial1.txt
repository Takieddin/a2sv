class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        left, right = 0, 0
        while left < len(nums) and right < len(nums):
            if nums[left] != 0 and nums[right] != 0:
                left += 1
                right += 1
                continue
            if nums[right] == 0:
                right += 1
                continue
            nums[left] = nums[right]
            nums[right] = 0
            left += 1
