class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        if not nums:
            return 0.0
        max_sum = -inf
        sume = 0
        l = 0
        for r in range(len(nums)):
            sume += nums[r]
            if r - l + 1 > k:
                sume -= nums[l]
                l += 1
            if r - l + 1 == k:
                max_sum = max(max_sum, sume)
        return max_sum / k
