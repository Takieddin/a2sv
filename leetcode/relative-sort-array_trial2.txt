class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        hmap = {}
        for i in range(len(arr2)):
            hmap[arr2[i]] = i
        def sort_key(x):
            if x in hmap:
                return hmap[x]
            return x + len(arr2)
        arr1.sort(key = sort_key)
        return arr1
