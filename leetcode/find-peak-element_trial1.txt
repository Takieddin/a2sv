class Solution:
    def findPeakElement(self, nums: List[int]) -> int:
        n = len(nums)
        l, r = 0 ,n
        def get(i):
            if i < 0 or i >= n:
                return -inf
            return nums[i]
        while l <= r:
            m = (l + r) // 2
            if get(m - 1) < get(m) > get(m + 1):
                return m
            elif get(m - 1) > get(m):
                r = m - 1
            else:
                l = m + 1
