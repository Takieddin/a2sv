# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseBetween(self, head: Optional[ListNode], left: int, right: int) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        left_p = pleft = right_p = aright = dummy
        for i in range(left - 1):
            pleft = pleft.next
        left_p = pleft.next
        for i in range(right):
            right_p = right_p.next
        aright = right_p.next
        right_p.next = None

        begin, end = left_p, right_p
        curr = begin
        prev = None
        while curr:
            tmp = curr.next
            curr.next = prev
            prev = curr
            curr = tmp
        begin.next = aright
        pleft.next = end
        return dummy.next
