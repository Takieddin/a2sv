class Node:
    def __init__(self, key, val):
        self.key = key
        self.val = val
        self.next = None
        self.prev = None

class LRUCache:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.head = Node(-1, -1)
        self.tail = Node(-1, -1)
        self.count = 0
        self.hashm = {}
        self.head.next = self.tail
        self.tail.prev = self.head
    def getNode(self, key: int) -> int:
        if not key in self.hashm:
            return None
        node = self.hashm[key]
        node.prev.next = node.next
        node.next.prev = node.prev
        node.next = self.head.next
        node.prev = self.head
        self.head.next.prev = node
        self.head.next = node
        return node
    def get(self, key):
        node = self.getNode(key)
        return node.val if node else -1       

    def put(self, key: int, value: int) -> None:
        node = self.getNode(key)
        if node:
            node.val = value
        else:
            node = Node(key, value)
            node.next = self.head.next
            node.prev = self.head
            self.hashm[key] = node
            self.head.next.prev = node
            self.head.next = node
            if self.count < self.capacity:
                self.count += 1
            else:
                to_del = self.tail.prev
                del self.hashm[to_del.key]
                self.tail.prev = to_del.prev
                to_del.prev.next = to_del.next
