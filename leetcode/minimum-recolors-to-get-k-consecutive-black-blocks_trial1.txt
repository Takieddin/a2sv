class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        window = Counter(blocks[:k])
        res = window["W"]
        for r in range(k, len(blocks)):
            window[blocks[r]] += 1
            window[blocks[r - k]] -= 1
            res = min(res, window["W"])
        return res
