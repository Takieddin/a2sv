class Solution:
    def simplifyPath(self, path: str) -> str:
        arr = path.split("/")
        stack = ["/"]
        for e in arr:
            if e and e != ".":
                if e == "..":
                    if len(stack) >= 3:
                        stack.pop()
                        stack.pop()
                else:
                    stack.append(e)
                    stack.append("/")
        if len(stack) >= 3:
            stack.pop()
        return "".join(stack)
