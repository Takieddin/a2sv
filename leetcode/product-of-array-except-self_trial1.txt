class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        t = 1
        left = [t := t * num for num in nums]
        t = 1
        right = [t := t * num for num in nums[::-1]][::-1]
        ans = [0] * len(nums)
        ans[0], ans[-1] = right[1], left[-2]
        for i in range(1, len(nums) - 1):
            ans[i] = right[i + 1] * left[i - 1]

        return ans
