class RecentCounter:

    def __init__(self):
        from collections import deque
        self.records = deque()
        

    def ping(self, t: int) -> int:
        self.records.append(t)
        while self.records and self.records[0] < t - 3000:
            self.records.popleft()
        return len(self.records)
