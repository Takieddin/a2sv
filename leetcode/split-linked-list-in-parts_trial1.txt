# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def splitListToParts(self, head: Optional[ListNode], k: int) -> List[Optional[ListNode]]:
        n = 0
        dummy = ListNode(-1, head)
        ptr = dummy
        while ptr and ptr.next:
            ptr = ptr.next
            n += 1
        res = [None] * k
        remainder = n % k
        q = n // k
        for i in range(k):
            ptr = dummy
            num_of_nodes = q + int(i < remainder)
            res[i] = ptr.next
            for _ in range(num_of_nodes):
                ptr = ptr.next
            tmp = ptr
            if ptr:
                ptr = ptr.next
                dummy.next = ptr
                tmp.next = None
        return res
