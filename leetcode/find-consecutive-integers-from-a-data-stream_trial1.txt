class DataStream:

    def __init__(self, value: int, k: int):
        self.value = value
        self.k = k
        self.ctr = 0        

    def consec(self, num: int) -> bool:
        if num == self.value:
            self.ctr += 1
        else:
            self.ctr = 0
        return self.ctr >= self.k
