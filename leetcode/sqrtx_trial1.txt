class Solution:
    def mySqrt(self, x: int) -> int:
        l, r = 0, x
        while l <= r:
            m = (l + r) // 2
            m2 = m * m
            pm2 = (m + 1) * (m + 1)
            if m2 <= x < pm2:
                return m
            elif m2 > x:
                r = m - 1
            else :
                l = m + 1
