# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(-1, head)
        ptr = dummy
        while ptr and ptr.next and ptr.next.next:
            first = ptr.next
            second = first.next
            third = second.next
            ptr.next = second
            second.next = first
            first.next = third
            ptr = first
        return dummy.next
