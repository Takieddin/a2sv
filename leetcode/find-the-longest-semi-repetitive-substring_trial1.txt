class Solution:
    def longestSemiRepetitiveSubstring(self, s: str) -> int:
        adj_count = 0
        res = 1
        last_idx = 0
        l = 0
        prev = -1
        for r in range(1, len(s)):
            if s[r] == s[r -1]:
                if prev != -1:
                    l = prev

                prev = r
            res = max(res, r - l + 1)
        return res
