class Solution:
    def longestSemiRepetitiveSubstring(self, s: str) -> int:
        adj_count = 0
        res = 1
        last_idx = 0
        l = 0
        for r in range(1, len(s)):
            if s[r] == s[r -1]:
                adj_count += 1
            while adj_count == 2:
                if s[l] == s[l +1]:
                    adj_count -= 1
                l += 1
            res = max(res, r - l + 1)
        return res
