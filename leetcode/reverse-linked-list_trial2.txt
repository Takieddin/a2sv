# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        tail = None
        if not head:
            return head
        tmp = head.next
        while tmp:
            head.next = tail
            tail = head
            head = tmp
            tmp = tmp.next
            head.next = tail
        return head
