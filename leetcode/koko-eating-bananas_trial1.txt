class Solution:
    def minEatingSpeed(self,piles: List[int], H: int) -> int:
        def feasible(speed) -> bool:
            return sum(math.ceil(pile / speed)  for pile in piles) <= H  # slower        

        left, right = 1, max(piles) // (H // len(piles)) + 1
        res = -1
        while left <= right:
            mid = left  + (right - left) // 2
            if feasible(mid):
                res = mid
                right = mid - 1 
            else:
                left = mid + 1
        return res
