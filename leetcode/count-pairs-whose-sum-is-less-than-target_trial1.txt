class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        nums.sort()
        l, r = 0, len(nums) - 1
        res = 0
        while l < r:
            pair = nums[l] + nums[r]
            if pair < target:
                res += r - l
                l += 1
            else :
                r -= 1
        return res
