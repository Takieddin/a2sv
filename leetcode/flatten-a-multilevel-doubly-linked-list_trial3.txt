class Solution:
    def flatten(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if not head:
            return None
        stack = []
        ptr = head
        stack.append(head.next)
        stack.append(head.child)
        head.child = None
        head.next = None
        while stack:
            next_node = stack.pop()
            if next_node:
                ptr.next = next_node
                next_node.prev = ptr
                stack.append(next_node.next)
                stack.append(next_node.child)
                next_node.child = None
                next_node.next = None
                ptr = next_node
        return head
