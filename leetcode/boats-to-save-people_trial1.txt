class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:
        people.sort()
        res = 0
        n = len(people) 
        l, r = 0, n - 1
        while l <= r:
            print(l, r)
            if l == r:
                return res + 1
            if people[l] + people[r] <= limit:
                l += 1
            r -= 1
            res += 1
        return res
